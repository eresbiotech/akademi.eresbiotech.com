{{- $campaign := .campaign -}} {{- $course := .course -}} {{- $quota := int ($course.quota | default $campaign.default_quota | default 10) -}} {{- $remaining := -1 -}} {{- if isset $course "remaining_slots" -}} {{- $remaining = int $course.remaining_slots -}} {{- if lt $remaining 0 -}} {{- $remaining = 0 -}} {{- else if gt $remaining $quota -}} {{- $remaining = $quota -}} {{- end -}} {{- else -}} {{- $filledRaw := int ($course.filled | default 0) -}} {{- $filled := $filledRaw -}} {{- if lt $filledRaw 0 -}} {{- $filled = 0 -}} {{- else if gt $filledRaw $quota -}} {{- $filled = $quota -}} {{- end -}} {{- $remaining = sub $quota $filled -}} {{- end -}} {{- return $remaining -}}
